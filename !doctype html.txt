  <script>
    const YT_ID = 'qmtLI2RoZJo';  // Aquí ya está tu video

    const playBtn = document.getElementById('playBtn');
    const muteBtn = document.getElementById('muteBtn');
    const player = document.getElementById('player');

    function makeEmbed({mute=false,autoplay=true}){
      const m = mute ? '&mute=1' : '';
      return `<iframe src="https://www.youtube.com/embed/${YT_ID}?rel=0&controls=1&autoplay=${autoplay?1:0}&loop=1&playlist=${YT_ID}${m}" frameborder="0" allow="autoplay; encrypted-media; picture-in-picture" allowfullscreen></iframe>`;
    }

    playBtn.addEventListener('click', ()=>{
      if(!player.dataset.playing){
        player.innerHTML = makeEmbed({mute:false,autoplay:true});
        player.dataset.playing = '1';
        playBtn.textContent = 'Pausar música';
      } else {
        player.innerHTML = '';
        delete player.dataset.playing;
        playBtn.textContent = 'Reproducir música (Rauw Alejandro)';
      }
    });

    muteBtn.addEventListener('click', ()=>{
      if(!player.dataset.playing) return;
      const iframe = player.querySelector('iframe');
      if(!iframe) return;
      const muted = player.dataset.muted === '1';
      player.innerHTML = makeEmbed({mute: !muted, autoplay:true});
      player.dataset.muted = muted ? '0' : '1';
      muteBtn.textContent = muted ? 'Silenciar / Desmutear' : 'Desmutear / Silenciar';
    });
  </script>
